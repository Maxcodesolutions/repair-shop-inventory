You can also check data manually in the browser console:

// Check localStorage data
console.log('Inventory:', localStorage.getItem('inventory'));
console.log('Customers:', localStorage.getItem('customers'));
console.log('Users:', localStorage.getItem('users'));

// Check if data is valid JSON
try {
  const customers = JSON.parse(localStorage.getItem('customers'));
  console.log('Valid customers:', customers);
} catch(e) {
  console.error('Invalid customers data:', e);
}
